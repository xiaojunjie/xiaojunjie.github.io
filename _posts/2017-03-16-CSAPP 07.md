//main.c
void swap(); // UND
int data[] = {11,22}; // .data
int main() { // .text
    swap();
    static int a = 1; // .data
    return a;
}
//swap.c
extern int data[];  // UND
int * bufp0 = &data[0]; // .data
int * bufp1; // .bss
void swap() { // .text
    int temp; // stack
    bufp1 = &data[0];
    temp = * bufp0;
    * bufp0 = * bufp1;
    * bufp1 = temp;
}

{% endhighlight %}

### 链接器
- Rule 1: Multiple strong symbols are not allowed.
- Rule 2: Given a strong symbol and multiple weak symbols, choose the strong
symbol.
- Rule 3: Given multiple weak symbols, choose any of the weak symbols.
